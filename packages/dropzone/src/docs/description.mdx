import { Dropzone } from '..';

Компонент drag-and-drop контейнер для прикрепления файлов.

```tsx live
const Container = ({ children }) => (
    <div
        style={{
            width: 400,
            height: 200,
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
        }}
    >
        {children}
    </div>
);

const Component = () => {
    const [filesList, setFilesList] = React.useState([]);
    const handleDrop = files => {
        setFilesList(
            Array.from(files)
                .map(file => file.name)
                .filter(Boolean),
        );
    };
    return (
        <Dropzone onDrop={handleDrop}>
            <Container>
                {filesList.length > 0 ? (
                    <ul>
                        {filesList.map(file => (
                            <li key={file}>{file}</li>
                        ))}
                    </ul>
                ) : (
                    <span>children</span>
                )}
            </Container>
        </Dropzone>
    );
};

render(<Component />);
```
